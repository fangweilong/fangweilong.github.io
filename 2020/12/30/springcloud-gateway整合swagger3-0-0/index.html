	private static final String SWAGGER2URL = "/v3/api-docs";

	/**
	 * ç½‘å…³åº”ç”¨åç§°

	@Value("${spring.application.name}")
	private String self;

	@Autowired
	public MySwaggerResourcesProvider(RouteLocator routeLocator) {
		this.routeLocator = routeLocator;
	}

	/**
	 * å¯¹äºgatewayæ¥è¯´è¿™å—æ¯”è¾ƒé‡è¦ è®©swaggerèƒ½æ‰¾åˆ°å¯¹åº”çš„æœåŠ¡

	@Override
	public List<SwaggerResource> get() {
		List<SwaggerResource> resources = new ArrayList<>();
		List<String> routeHosts = new ArrayList<>();
		// è·å–æ‰€æœ‰å¯ç”¨çš„hostï¼šserviceId
		routeLocator.getRoutes().filter(route -> route.getUri().getHost() != null)
				.filter(route -> !self.equals(route.getUri().getHost()))
				.subscribe(route -> routeHosts.add(route.getUri().getHost()));

		// è®°å½•å·²ç»æ·»åŠ è¿‡çš„server
		Set<String> dealed = new HashSet<>();

				swaggerResource.setUrl(url);
				swaggerResource.setName(instance);

			}
		});
		return resources;
	}
